```dataviewjs
const container = dv.container;
container.classList.add("snc-root");

// ═══════════════════════════════════════════
// HEADER
// ═══════════════════════════════════════════
const header = container.createEl("div", { cls: "snc-header" });

const headerTop = header.createEl("div", { cls: "snc-header-top" });
headerTop.createEl("h1", { text: "FAMTEAM", cls: "snc-title" });

const now = new Date();
const hour = now.getHours();
const greeting = hour < 5 ? "Тишина ночи" : hour < 12 ? "Доброе утро" : hour < 18 ? "Глубокий фокус" : "Вечерний обзор";
const timeStr = now.toLocaleTimeString("ru-RU", { hour: "2-digit", minute: "2-digit" });
headerTop.createEl("span", { text: timeStr, cls: "snc-clock" });

header.createEl("div", { cls: "snc-divider" });

const dateStr = now.toLocaleDateString("ru-RU", { weekday: "long", day: "numeric", month: "long", year: "numeric" });
header.createEl("p", { text: `${greeting}. ${dateStr}`, cls: "snc-subtitle" });

// ═══════════════════════════════════════════
// STATS
// ═══════════════════════════════════════════
const projects = dv.pages('"Проекты"');
const allTasks = projects.file.tasks;
const openTasks = allTasks.where(t => !t.completed);
const doneTasks = allTasks.where(t => t.completed);
const inProgress = allTasks.where(t => !t.completed && t.text && t.text.includes("#in_progress"));
const activeProjects = projects.where(p => p.статус === "активен");

const stats = container.createEl("div", { cls: "snc-stats" });

function addStat(parent, value, label, accent) {
    const card = parent.createEl("div", { cls: "snc-stat" });
    const valEl = card.createEl("span", { text: String(value), cls: "snc-stat-value" });
    if (accent) valEl.style.color = accent;
    card.createEl("span", { text: label, cls: "snc-stat-label" });
}

addStat(stats, activeProjects.length, "Активных проектов", "#42be65");
addStat(stats, inProgress.length, "В работе", "#4589ff");
addStat(stats, openTasks.length, "Открыто задач");
addStat(stats, doneTasks.length, "Завершено");

// ═══════════════════════════════════════════
// КОМАНДА
// ═══════════════════════════════════════════
const teamSection = container.createEl("div", { cls: "snc-section" });
teamSection.createEl("h2", { text: "Команда", cls: "snc-section-title" });

const teamGrid = teamSection.createEl("div", { cls: "snc-team-grid" });

const members = [
    { name: "Сергей", role: "backend + Claude", tag: "#sergei", color: "#42be65" },
    { name: "Юля", role: "frontend + design", tag: "#yulia", color: "#be95ff" },
    { name: "Claude", role: "AI-партнёр", tag: "#claude", color: "#4589ff" }
];

members.forEach(m => {
    const mTasks = allTasks.where(t => !t.completed && t.text && t.text.includes(m.tag));
    const mActive = mTasks.where(t => t.text.includes("#in_progress")).length;
    const card = teamGrid.createEl("div", { cls: "snc-team-card" });
    const nameRow = card.createEl("div", { cls: "snc-team-name-row" });
    const dot = nameRow.createEl("span", { cls: "snc-dot" });
    dot.style.backgroundColor = m.color;
    nameRow.createEl("span", { text: m.name, cls: "snc-team-name" });
    card.createEl("span", { text: m.role, cls: "snc-team-role" });
    card.createEl("span", { text: `${mActive} в работе · ${mTasks.length} всего`, cls: "snc-team-stats" });
});

// ═══════════════════════════════════════════
// ПРОЕКТЫ
// ═══════════════════════════════════════════
const section = container.createEl("div", { cls: "snc-section" });
section.createEl("h2", { text: "Проекты", cls: "snc-section-title" });

const listHeader = section.createEl("div", { cls: "snc-row snc-row-header" });
listHeader.createEl("span", { text: "Название", cls: "snc-col snc-col-name" });
listHeader.createEl("span", { text: "Статус", cls: "snc-col snc-col-status" });
listHeader.createEl("span", { text: "Стек", cls: "snc-col snc-col-stack" });
listHeader.createEl("span", { text: "Прогресс", cls: "snc-col snc-col-progress" });
listHeader.createEl("span", { text: "Задачи", cls: "snc-col snc-col-tasks" });

const statusColors = { "активен": "#42be65", "R&D": "#f1c21b", "на паузе": "#525252" };

const sorted = projects.sort(p => p.приоритет ?? 99, "asc");

for (const page of sorted) {
    const pageTasks = page.file.tasks;
    const done = pageTasks.where(t => t.completed).length;
    const total = pageTasks.length;
    const pct = total > 0 ? Math.round((done / total) * 100) : 0;
    const status = page.статус ?? "—";

    const row = section.createEl("div", { cls: "snc-row" });

    // Название + приоритет
    const nameCol = row.createEl("span", { cls: "snc-col snc-col-name" });
    const nameWrap = nameCol.createEl("span", { cls: "snc-name-wrap" });
    nameWrap.createEl("span", { text: `#${page.приоритет ?? '—'}`, cls: "snc-priority" });
    const link = nameWrap.createEl("a", { text: page.file.name, cls: "snc-link", href: page.file.path });
    link.addEventListener("click", e => { e.preventDefault(); app.workspace.openLinkText(page.file.path, ""); });

    // Статус
    const statusCol = row.createEl("span", { cls: "snc-col snc-col-status" });
    const badge = statusCol.createEl("span", { text: status, cls: "snc-badge" });
    badge.style.borderColor = statusColors[status] || "#525252";
    badge.style.color = statusColors[status] || "#8d8d8d";

    // Стек
    const stackCol = row.createEl("span", { cls: "snc-col snc-col-stack" });
    const stackStr = page.стек || "";
    const stackItems = stackStr.split(",").map(s => s.trim()).filter(Boolean).slice(0, 3);
    stackItems.forEach(s => stackCol.createEl("span", { text: s, cls: "snc-stack-tag" }));

    // Прогресс
    const progCol = row.createEl("span", { cls: "snc-col snc-col-progress" });
    const barOuter = progCol.createEl("div", { cls: "snc-bar-outer" });
    const barInner = barOuter.createEl("div", { cls: "snc-bar-inner" });
    barInner.style.width = `${pct}%`;
    if (statusColors[status]) barInner.style.background = statusColors[status];
    progCol.createEl("span", { text: `${pct}%`, cls: "snc-bar-label" });

    // Задачи
    row.createEl("span", { text: `${done}/${total}`, cls: "snc-col snc-col-tasks" });
}

// ═══════════════════════════════════════════
// СЕЙЧАС В РАБОТЕ
// ═══════════════════════════════════════════
const activeSection = container.createEl("div", { cls: "snc-section" });
activeSection.createEl("h2", { text: "Сейчас в работе", cls: "snc-section-title" });

const activeTasks = allTasks.where(t => !t.completed && t.text && t.text.includes("#in_progress"));

const assigneeMap = {
    "#sergei": { label: "С", color: "#42be65" },
    "#yulia": { label: "Ю", color: "#be95ff" },
    "#claude": { label: "AI", color: "#4589ff" }
};

if (activeTasks.length === 0) {
    activeSection.createEl("p", { text: "Нет активных задач.", cls: "snc-empty" });
} else {
    const taskList = activeSection.createEl("div", { cls: "snc-task-list" });

    for (const task of activeTasks) {
        const item = taskList.createEl("div", { cls: "snc-task-item" });

        // Определяем исполнителя
        let assignee = null;
        for (const [tag, info] of Object.entries(assigneeMap)) {
            if (task.text.includes(tag)) { assignee = info; break; }
        }

        const dot = item.createEl("span", { cls: "snc-dot" });
        dot.style.backgroundColor = assignee ? assignee.color : "#8d8d8d";

        // Чистый текст задачи
        const cleanText = task.text.replace(/#\w+/g, "").trim();
        item.createEl("span", { text: cleanText, cls: "snc-task-text" });

        // Проект
        const project = task.path.split('/').slice(-2, -1)[0] || '';
        item.createEl("span", { text: project, cls: "snc-task-project" });

        // Исполнитель
        if (assignee) {
            const badge = item.createEl("span", { text: assignee.label, cls: "snc-task-assignee" });
            badge.style.borderColor = assignee.color;
            badge.style.color = assignee.color;
        }

        item.addEventListener("click", () => app.workspace.openLinkText(task.path, ""));
    }
}

// ═══════════════════════════════════════════
// ОЧЕРЕДЬ
// ═══════════════════════════════════════════
const queueSection = container.createEl("div", { cls: "snc-section" });
queueSection.createEl("h2", { text: "Очередь", cls: "snc-section-title" });

const queueTasks = allTasks
    .where(t => !t.completed && t.text && !t.text.includes("#in_progress"))
    .limit(10);

if (queueTasks.length === 0) {
    queueSection.createEl("p", { text: "Очередь пуста.", cls: "snc-empty" });
} else {
    const qList = queueSection.createEl("div", { cls: "snc-task-list" });
    for (const task of queueTasks) {
        const item = qList.createEl("div", { cls: "snc-task-item" });

        let assignee = null;
        for (const [tag, info] of Object.entries(assigneeMap)) {
            if (task.text.includes(tag)) { assignee = info; break; }
        }

        const dot = item.createEl("span", { cls: "snc-dot" });
        dot.style.backgroundColor = assignee ? assignee.color : "#8d8d8d";

        const cleanText = task.text.replace(/#\w+/g, "").trim();
        item.createEl("span", { text: cleanText, cls: "snc-task-text" });

        const project = task.path.split('/').slice(-2, -1)[0] || '';
        item.createEl("span", { text: project, cls: "snc-task-project" });

        item.addEventListener("click", () => app.workspace.openLinkText(task.path, ""));
    }
}

// ═══════════════════════════════════════════
// ПОСЛЕДНИЕ ИЗМЕНЕНИЯ
// ═══════════════════════════════════════════
const recentSection = container.createEl("div", { cls: "snc-section" });
recentSection.createEl("h2", { text: "Последние изменения", cls: "snc-section-title" });

const recentGrid = recentSection.createEl("div", { cls: "snc-recent-grid" });

const recentFiles = dv.pages('"Проекты" OR "Идеи" OR "Команда"')
    .sort(p => p.file.mtime, "desc")
    .limit(6);

for (const f of recentFiles) {
    const card = recentGrid.createEl("div", { cls: "snc-recent-card" });
    const link = card.createEl("a", { text: f.file.name, cls: "snc-link", href: f.file.path });
    link.addEventListener("click", e => { e.preventDefault(); app.workspace.openLinkText(f.file.path, ""); });

    const timeAgo = Math.floor((Date.now() - new Date(f.file.mtime).getTime()) / 60000);
    const timeLabel = timeAgo < 60 ? `${timeAgo} мин` : timeAgo < 1440 ? `${Math.floor(timeAgo/60)} ч` : `${Math.floor(timeAgo/1440)} дн`;
    card.createEl("span", { text: timeLabel, cls: "snc-recent-time" });
}
```
