    const valEl = card.createEl("span", { text: String(value), cls: "snc-stat-value" });
    if (accent) valEl.style.color = accent;
    card.createEl("span", { text: label, cls: "snc-stat-label" });
}

addStat(stats, activeProjects.length, "Активных проектов", "#42be65");

        type: "text",

        style: "flex: 1; padding: 12px 16px; border: 1px solid var(--snc-border); background: var(--snc-surface); color: var(--snc-text); font-family: var(--snc-font); font-size: 14px; outline: none;"

captureInput.addEventListener("focus", () => captureInput.style.borderColor = "var(--snc-accent)");
captureInput.addEventListener("blur", () => captureInput.style.borderColor = "var(--snc-border)");

captureInput.addEventListener("keydown", async (e) => {
    if (e.key === "Enter" && captureInput.value.trim()) {
        const text = captureInput.value.trim();
        const now = new Date();
        const time = now.toLocaleTimeString("ru-RU", { hour: "2-digit", minute: "2-digit" });
        const date = now.toLocaleDateString("ru-RU");
        const line = `\n- [ ] ${text} — *${date} ${time}*`;

        const file = app.vault.getAbstractFileByPath("Идеи/Входящие.md");
        if (file) {
            const content = await app.vault.read(file);
            await app.vault.modify(file, content + line);
        }

        captureInput.value = "";
        captureInput.placeholder = "✅ Сохранено! Ещё идея...";
        setTimeout(() => captureInput.placeholder = "Идея, задача, заметка... Enter чтобы сохранить", 2000);
    }
});
